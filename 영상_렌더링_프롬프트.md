# 영상 렌더링 기능 - API 및 구현 가이드

## 📋 개요
이미지, 오디오, 자막을 합쳐서 최종 영상을 렌더링하는 기능입니다. FFmpeg를 사용하여 서버 사이드에서 렌더링합니다.

---

## 🔌 API 정보

### 기본 정보
- **API 엔드포인트**: `POST /render`
- **HTTP 메서드**: `POST`
- **Content-Type**: `application/json`
- **구현**: Python Flask (FFmpeg 사용)

---

## 📝 요청 형식

### 요청 Body

```json
{
  "audioBase64": "base64_encoded_audio",
  "audioGcsUrl": "https://storage.googleapis.com/...",
  "subtitles": [
    {
      "id": 1,
      "start": 0.0,
      "end": 2.5,
      "text": "첫 번째 자막"
    },
    {
      "id": 2,
      "start": 2.5,
      "end": 5.0,
      "text": "두 번째 자막"
    }
  ],
  "showSubtitles": true,
  "characterImage": "data:image/png;base64,...",
  "characterImageGcsUrl": "https://storage.googleapis.com/...",
  "characterImageUrl": "https://example.com/video.mp4",
  "autoImages": [
    {
      "id": "1",
      "url": "https://example.com/image1.jpg",
      "startTime": 0.0,
      "endTime": 10.0,
      "isVideo": false
    },
    {
      "id": "2",
      "url": "https://example.com/video2.mp4",
      "startTime": 10.0,
      "endTime": 20.0,
      "isVideo": true
    }
  ],
  "duration": 120.0,
  "config": {
    "width": 1920,
    "height": 1080,
    "fps": 30
  }
}
```

### 필수 파라미터

- **오디오**: `audioBase64` 또는 `audioGcsUrl` 중 하나 필수
- **배경 이미지/비디오**: `characterImage`, `characterImageGcsUrl`, 또는 `characterImageUrl` 중 하나 필수

### 선택 파라미터

- `subtitles`: 자막 배열 (선택)
- `showSubtitles`: 자막 표시 여부 (기본값: true)
- `autoImages`: 추가 이미지/비디오 배열 (선택)
- `duration`: 영상 길이 (초)
- `config`: 영상 설정 (기본값: 1920x1080, 30fps)

### 2. 이미지/비디오 처리

```python
def download_image(img_url, output_path):
    """이미지를 다운로드하여 PNG로 변환"""
    from PIL import Image
    import io
    
    if img_url.startswith("data:"):
        # data URL 처리
        header, b64data = img_url.split(",", 1)
        img_data = base64.b64decode(b64data)
    else:
        # HTTP/HTTPS URL 처리
        img_response = requests.get(img_url, timeout=30)
        img_data = img_response.content
    
    # PIL로 열어서 PNG로 변환
    img = Image.open(io.BytesIO(img_data))
    if img.mode in ('RGBA', 'LA', 'P'):
        # 투명도가 있는 경우 흰색 배경에 합성
        rgb_img = Image.new('RGB', img.size, (255, 255, 255))
        if img.mode == 'P':
            img = img.convert('RGBA')
        rgb_img.paste(img, mask=img.split()[-1] if img.mode in ('RGBA', 'LA') else None)
        img = rgb_img
    elif img.mode != 'RGB':
        img = img.convert('RGB')
    
    # PNG로 저장
    full_path = f"{output_path}.png"
    img.save(full_path, 'PNG', optimize=True)
    return full_path
```

## 📋 FFmpeg 설정

### 비디오 코덱 설정

```python
output_kwargs = {
    'vcodec': 'libx264',      # H.264 코덱
    'preset': 'ultrafast',    # 인코딩 속도 (ultrafast, fast, medium, slow 등)
    'crf': 28,                # 품질 (23=고품질, 28=중간품질)
    'tune': 'stillimage',     # 정적 이미지 최적화
    'pix_fmt': 'yuv420p',     # 픽셀 포맷 (호환성)
    'threads': 0,             # 모든 CPU 코어 사용
    'acodec': 'aac',          # 오디오 코덱
    'movflags': '+faststart'  # 스트리밍 최적화
}
```

## 📋 구현 예시 (Python Flask)

```python
from flask import Flask, request, jsonify
import ffmpeg
import subprocess
import tempfile
import base64
import requests
import json
import os

app = Flask(__name__)

@app.route('/render', methods=['POST'])
def render_video():
    try:
        data = request.json
        
        # 요청 데이터 검증
        audio_base64 = data.get('audioBase64')
        audio_gcs_url = data.get('audioGcsUrl')
        subtitles = data.get('subtitles', [])
        show_subtitles = data.get('showSubtitles', True)
        character_image = data.get('characterImage')
        character_image_gcs_url = data.get('characterImageGcsUrl')
        character_image_url = data.get('characterImageUrl')
        auto_images = data.get('autoImages', [])
        duration = data.get('duration', 0)
        config = data.get('config', {'width': 1920, 'height': 1080, 'fps': 30})
        
        # 필수 파라미터 검증
        if not audio_base64 and not audio_gcs_url:
            return jsonify({"success": False, "error": "오디오가 필요합니다."}), 400
        
        if not character_image and not character_image_gcs_url and not character_image_url:
            return jsonify({"success": False, "error": "배경 이미지가 필요합니다."}), 400
        
        # 임시 디렉토리 생성
        temp_dir = tempfile.mkdtemp()
        
        try:
            # 1. 오디오 처리
            if audio_gcs_url:
                audio_response = requests.get(audio_gcs_url, timeout=300)
                audio_data = audio_response.content
            else:
                audio_data = base64.b64decode(audio_base64)
            
            audio_path = f"{temp_dir}/audio.wav"
            with open(audio_path, 'wb') as f:
                f.write(audio_data)
            
            # 실제 오디오 길이 측정
            probe_cmd = [
                'ffprobe', '-v', 'quiet', '-print_format', 'json',
                '-show_format', audio_path
            ]
            probe_result = subprocess.run(probe_cmd, capture_output=True, text=True)
            probe_data = json.loads(probe_result.stdout)
            actual_audio_duration = float(probe_data.get('format', {}).get('duration', 0))
            
            # 2. 배경 이미지 처리
            if character_image_gcs_url:
                img_path = download_image(character_image_gcs_url, f"{temp_dir}/background")
            elif character_image_url:
                # 비디오인 경우
                video_response = requests.get(character_image_url, timeout=300, stream=True)
                video_path = f"{temp_dir}/background_video.mp4"
                with open(video_path, 'wb') as f:
                    for chunk in video_response.iter_content(chunk_size=8192):
                        if chunk:
                            f.write(chunk)
                img_path = video_path
            else:
                img_path = download_image(character_image, f"{temp_dir}/background")
            
            # 3. 자막 처리
            srt_path = None
            if show_subtitles and subtitles:
                duration_ratio = actual_audio_duration / duration if duration > 0 else 1.0
                srt_content = generate_srt(subtitles, duration_ratio)
                srt_path = f"{temp_dir}/subtitles.srt"
                with open(srt_path, 'w', encoding='utf-8') as f:
                    f.write(srt_content)
            
            # 4. FFmpeg 렌더링
            output_path = f"{temp_dir}/output.mp4"
            
            # 입력 설정
            if character_image_url and character_image_url.endswith('.mp4'):
                # 비디오 배경
                input_video = ffmpeg.input(img_path, t=actual_audio_duration)
                video_stream = input_video['v']
            else:
                # 이미지 배경
                input_image = ffmpeg.input(img_path, loop=1, t=actual_audio_duration)
                video_stream = input_image['v']
            
            input_audio = ffmpeg.input(audio_path)
            
            # 자막 필터 추가
            if srt_path:
                video_stream = ffmpeg.filter(
                    video_stream,
                    'subtitles',
                    srt_path,
                    force_style="FontName=Noto Sans CJK KR,FontSize=24,PrimaryColour=&Hffffff,OutlineColour=&H000000,Outline=2,Alignment=2,MarginV=50"
                )
            
            # 출력 설정
            stream = ffmpeg.output(
                video_stream,
                input_audio['a'],
                output_path,
                vcodec='libx264',
                preset='ultrafast',
                crf=28,
                tune='stillimage',
                pix_fmt='yuv420p',
                threads=0,
                acodec='aac',
                t=actual_audio_duration,
                movflags='+faststart'
            )
            
            ffmpeg.run(stream, overwrite_output=True, quiet=True)
            
            # 5. 결과 반환
            with open(output_path, 'rb') as f:
                video_data = f.read()
                video_base64 = base64.b64encode(video_data).decode('utf-8')
            
            return jsonify({
                "success": True,
                "videoBase64": video_base64,
                "duration": actual_audio_duration
            })
            
        finally:
            # 임시 파일 정리
            import shutil
            shutil.rmtree(temp_dir, ignore_errors=True)
            
    except Exception as e:
        return jsonify({
            "success": False,
            "error": f"렌더링 중 오류 발생: {str(e)}"
        }), 500
```

## ⚠️ 주의사항

1. **오디오 길이**: 실제 오디오 길이를 측정하여 영상 길이를 결정합니다.
2. **자막 타이밍**: 실제 오디오 길이에 맞춰 자막 타이밍을 조정합니다.
3. **이미지 형식**: WebP 등 일부 형식은 PNG로 변환해야 합니다.
4. **메모리 관리**: 큰 파일은 스트리밍 방식으로 처리합니다.
5. **타임아웃**: 오디오/비디오 다운로드는 충분한 타임아웃을 설정합니다.
